\section{Preliminaries}

In this note, we are going to consider two scalar particles $\phi$ and $\varphi$.
The mass of $\phi$ is $m_\phi > 0$, while $\varphi$ is massless.
These two scalar particles interact with each other via the interaction term of
\begin{equation}
    \Lagrangian[\mathrm{int}][\phi, \varphi] = \frac{\lambda}{2} \phi \varphi^2
\end{equation}
with $\lambda$ being the coupling parameter that has the mass dimension of one.

We are going to consider the initial distribution of $\phi$ as
\begin{equation}
    f_\phi\qty(t_\ini, \rmp_\phi) = \frac{\twopi[2]}{2 \rmp_\phi^2} n_\phi^\upini \delta(\rmp_\phi - \epsilon)
    \label{eq:initial-condition-for-f_phi}
\end{equation}
with $\epsilon > 0$ being an infinitesimal to regularize the behavior of $f_\phi$ at $\rmp_\phi \to 0^+$.
It is easy to verify that
\begin{equation}
    \int \frac{\dd[3]{\vb{p}_\phi}}{\twopi[3]} f_\phi\qty(t_\ini, \rmp_\phi) = \int_0^\infty \frac{\rmp_\phi^2 \dd{\rmp_\phi}}{2 \pi^2} \frac{2 \pi^2}{\rmp_\phi^2} \delta(\rmp_\phi - \epsilon) = n_\phi^\upini.
\end{equation}

The Boltzmann transport equation (BTE) for $f_\phi$ reads
\begin{equation}
    \pdv{f_\phi}{t} - H \rmp_\phi \pdv{f_\phi}{\rmp_\phi} = \CT[f_\phi][][t, \rmp_\phi],
\end{equation}
where $H$ is the Hubble parameter and $\CT[f_\phi]$ is the collision term for $f_\phi$.
For $f_\varphi$, the BTE reads
\begin{equation}
    \pdv{f_\varphi}{t} - H \rmp_\varphi \pdv{f_\varphi}{\rmp_\varphi} = \CT[f_\varphi][][t, \rmp_\varphi].
\end{equation}

In the collision term, the factor consisting of distribution functions is presented as
\begin{equation}
    \begin{aligned}
        F_\pm(t, \rmp_\phi, \rmk_1, \rmk_2) & & \coloneq & & & \pm \qty{
            \begin{aligned}
                f_\phi\qty(t, \rmp_\phi) \qty[1 + f_\varphi\qty(t, \rmk_1)] \qty[1 + f_\varphi\qty(t, \rmk_2)] \qty[1 + f_h\qty(t, \rmk_h)] \\
                {} - f_\varphi\qty(t, \rmk_1) f_\varphi\qty(t, \rmk_2) f_h\qty(t, \rmk_h) \qty[1 + f_\phi\qty(t, \rmp_\phi)]
            \end{aligned}
        } \\
        & & = & & & \pm \qty{
            \begin{aligned}
                f_\phi\qty(t, \rmp_\phi) \qty[1 + f_\varphi\qty(t, \rmk_1) + f_\varphi\qty(t, \rmk_2) + f_\varphi\qty(t, \rmk_1) f_\varphi\qty(t, \rmk_2)] \\
                {} + f_h\qty(t, \rmk_h) \qty{f_\varphi\qty(t, \rmk_1) f_\varphi\qty(t, \rmk_2) - f_\phi\qty(t, \rmp_\phi) \qty[1 + f_\varphi\qty(t, \rmk_1) + f_\varphi\qty(t, \rmk_2)]}
            \end{aligned}
        } \\
        & & \simeq & & & \pm f_\phi\qty(t, \rmp_\phi) \qty[1 + f_\varphi\qty(t, \rmk_1) + f_\varphi\qty(t, \rmk_2) + f_\varphi\qty(t, \rmk_1) f_\varphi\qty(t, \rmk_2)],
    \end{aligned}
\end{equation}
where $\rmk_h = \sqrt{\rmp_\phi^2 + m_\phi^2} - \rmk_1 - \rmk_2$ is applied and we take the assumption of $f_h \ll 1$ in the last step.

\section{\boldmath \texorpdfstring{$\phi \leftrightarrow \varphi + \varphi + h$}{phi <-> varphi + varphi + h}}

In this section,  we would like to consider the process of
\begin{equation}
    \phi(p_\phi) \leftrightarrow \varphi(k_1) + \varphi(k_2) + h(k_h),
\end{equation}
whose norm squared of amplitude that summed over the left- and right-hand side degrees of freedom is given by
\begin{equation}
    \sumAmpNormSqr{\phi \to \varphi \varphi h} = \frac{2 \lambda^2}{\MPl^2} \qty(1 - \frac{m_\phi}{2 E_h})^2,
\end{equation}

\subsection{\boldmath For \texorpdfstring{$f_h$}{f\_h}}

The collision term for $f_h$ to this process reads
\begin{equation}
    \CT[f_h][\phi \leftrightarrow \varphi \varphi h][t, \rmk_h] = \frac{1}{2 E_h} \int \tildedp{p_\phi} ~ \tildedp{k_1} ~ \tildedp{k_2} ~ \twopidelta[4]{p_\phi - k_1 - k_2 - k_h} ~ \frac{1}{2} \sumAmpNormSqr{\phi \to \varphi \varphi h} F_+\qty(t, \rmp_\phi, \rmk_1, \rmk_2).
\end{equation}
Inserting two identities of
\begin{align}
    1 & \equiv \int \frac{\dd[4]{k_{12}}}{\twopi[4]} \twopidelta[4]{k_{12} - k_1 - k_2}, \\
    1 & \equiv \int \frac{\dd{s_{12}}}{\twopi} \twopidelta{s_{12} + k_{12}^2},
\end{align}
we have then
\begin{equation}
    \begin{aligned}
        & & & \CT[f_h][\phi \leftrightarrow \varphi \varphi h][t, \rmk_h] \\
        & = & & \frac{1}{2 \rmk_h} \int \frac{\dd{s_{12}}}{\twopi} \int \frac{\dd[4]{k_{12}}}{\twopi[4]} \twopidelta{s_{12} + k_{12}^2} \int \tildedp{p_\phi} ~ \tildedp{k_1} ~ \tildedp{k_2} ~ \twopidelta[4]{p_\phi - k_{12} - k_h} \twopidelta[4]{k_{12} - k_1 - k_2} \\
        & & & \qquad {} \times \frac{1}{2} \sumAmpNormSqr{\phi \to \varphi \varphi h} F_+\qty(t, \rmp_\phi, \rmk_1, \rmk_2) \\
        & = & & \frac{1}{2 \rmk_h} \int_0^\infty \frac{\dd{s_{12}}}{\twopi} \int \tildedp{p_\phi} ~ \tildedp{k_{12}} ~ \twopidelta[4]{p_\phi - k_{12} - k_h} \int \tildedp{k_1} ~ \tildedp{k_2} ~ \twopidelta[4]{k_{12} - k_1 - k_2} \\
        & & & \qquad {} \times \frac{1}{2} \sumAmpNormSqr{\phi \to \varphi \varphi h} F_+\qty(t, \rmp_\phi, \rmk_1, \rmk_2).
    \end{aligned}
\end{equation}
First, we have
\begin{equation}
    \begin{aligned}
        & & & \int \tildedp{p_\phi} ~ \tildedp{k_{12}} ~ \twopidelta[4]{p_\phi - k_{12} - k_h} \\
        & = & & \int \eval{\frac{\tildedp{p_\phi}}{2 E_{12}} \twopidelta{E_\phi - E_{12} - \rmk_h}}_{E_{12} = \sqrt{s_{12} + \rmp_\phi^2 + \rmk_h^2 - 2 \rmp_\phi \rmk_h \cos \expval{\rmp_\phi, \rmk_h}}} \\
        & = & & \int_0^\infty \frac{\rmp_\phi^2 \dd{\rmp_\phi}}{\twopi[2] 2 E_\phi} \frac{\twopi}{2 E_{12}} \frac{2 E_{12}}{2 \rmp_\phi \rmk_h} \tilde{\Theta}\qty(\rmp_\phi \ge \abs{\frac{\rmk_h m_\phi^2}{m_\phi^2 - s_{12}} - \frac{m_\phi^2 - s_{12}}{4 \rmk_h}}) \tilde{\Theta}\qty(s_{12} \le m_\phi^2) \\
        & = & & \frac{\tilde{\Theta}\qty(s_{12} \le m_\phi^2)}{4 \twopi[] \rmk_h} \int_{E_\phi^{(\min)}}^\infty \dd{E_\phi}
    \end{aligned}
\end{equation}
with
\begin{equation}
    E_\phi^{(\min)} = \frac{\rmk_h m_\phi^2}{m_\phi^2 - s_{12}} + \frac{m_\phi^2 - s_{12}}{4 \rmk_h}.
\end{equation}
And then, we also have
\begin{equation}
    \begin{aligned}
        \int \tildedp{k_1} ~ \tildedp{k_2} ~ \twopidelta[4]{k_{12} - k_1 - k_2} & = \int \eval{\frac{\tildedp{k_1}}{2 \rmk_2} \twopidelta{E_{12} - \rmk_1 - \rmk_2}}_{\rmk_2 = \sqrt{\rmk_{12}^2 + \rmk_1^2 - 2 \rmk_{12} \rmk_1 \cos \expval{\vb{k}_{12}, \vb{k}_1}}} \\
        & = \int_0^\infty \frac{\rmk_1^2 \dd{\rmk_1}}{\twopi[2] 2 \rmk_1} \frac{\twopi}{2 \rmk_2} \frac{2 \rmk_2}{2 \rmk_{12} \rmk_1} \tilde{\Theta}\qty(\frac{E_{12} - \rmk_{12}}{2} \le \rmk_1 \le \frac{E_{12} + \rmk_{12}}{2}) \\
        & = \frac{1}{4 \twopi[] \rmk_{12}} \int_{(E_{12} - \rmk_{12}) / 2}^{(E_{12} + \rmk_{12}) / 2} \dd{\rmk_1}.
    \end{aligned}
\end{equation}
Therefore, the collision term becomes
\begin{equation}
    \begin{aligned}
        \CT[f_h][\phi \leftrightarrow \varphi \varphi h][t, \rmk_h] & = & & \frac{1}{64 \twopi[3] \rmk_h^2} \int_0^{m_\phi^2} \dd{s_{12}} \int_{E_\phi^{(\min)}}^\infty \frac{\dd{E_\phi}}{\sqrt{\qty(E_\phi - \rmk_h)^2 - s_{12}}} \\
        & & & \qquad {} \times \int_{\qty(E_\phi - \rmk_h - \sqrt{\qty(E_\phi - \rmk_h)^2 - s_{12}}) / 2}^{\qty(E_\phi - \rmk_h + \sqrt{\qty(E_\phi - \rmk_h)^2 - s_{12}}) / 2} \dd{\rmk_1} \sumAmpNormSqr{\phi \to \varphi \varphi h} F_+\qty(t, \rmp_\phi, \rmk_1, \rmk_2).
    \end{aligned}
\end{equation}

Notice that $\sumAmpNormSqr{\phi \to \varphi \varphi h} \propto \lambda^2 \MPl^{-2} \ll 1$, we can expect that $f_h \propto \lambda^2 \MPl^{-2} \ll 1$, \ie,
\begin{equation}
    \begin{aligned}
        \CT[f_h][\phi \leftrightarrow \varphi \varphi h][t, \rmk_h] & \simeq & & \frac{1}{64 \twopi[3] \rmk_h^2} \int_0^{m_\phi^2} \dd{s_{12}} \int_{E_\phi^{(\min)}}^\infty \frac{\dd{E_\phi}}{\sqrt{\qty(E_\phi - \rmk_h)^2 - s_{12}}} \\
        & & & \qquad {} \times \int_{\qty(E_\phi - \rmk_h - \sqrt{\qty(E_\phi - \rmk_h)^2 - s_{12}}) / 2}^{\qty(E_\phi - \rmk_h + \sqrt{\qty(E_\phi - \rmk_h)^2 - s_{12}}) / 2} \dd{\rmk_1} \sumAmpNormSqr{\phi \to \varphi \varphi h} \\
        & & & \qquad {} \times f_\phi\qty(t, \rmp_\phi) \qty[1 + f_\varphi\qty(t, \rmk_1) + f_\varphi\qty(t, \rmk_2) + f_\varphi\qty(t, \rmk_1) f_\varphi\qty(t, \rmk_2)] + \order{\frac{\lambda^2}{\MPl^2}}.
    \end{aligned}
\end{equation}
Therefore, the assumptions of
\begin{align}
    f_\phi\qty(t, \rmp_\phi) & = \frac{\twopi[2]}{2 \rmp_\phi^2} n_\phi(t) \delta\qty(\rmp_\phi - \epsilon), \\
    f_\varphi\qty(t, \rmp_\varphi) & = \frac{2 \pi^2}{\rmp_\varphi^2} n_\varphi(t) \delta\qty(\rmp_\varphi - \frac{m_\phi}{2} \frac{a_\ini}{a(t)})
\end{align}
are hold up to $\order{\lambda^2 \MPl^{-2}}$, too.
The collision term can be further simplified as
\begin{equation}
    \begin{aligned}
        \CT[f_h][\phi \leftrightarrow \varphi \varphi h][t, \rmk_h] & & \simeq & & & \frac{1}{64 \twopi[3] \rmk_h^2} \int_0^{m_\phi^2} \dd{s_{12}} \int_{E_\phi^{(\min)}}^\infty \frac{\dd{E_\phi}}{\sqrt{\qty(E_\phi - \rmk_h)^2 - s_{12}}} \frac{\twopi[2]}{2 \rmp_\phi^2} n_\phi(t) \delta\qty(\rmp_\phi - \epsilon) \\
        & & & & & \qquad {} \times \int_{\qty(E_\phi - \rmk_h - \sqrt{\qty(E_\phi - \rmk_h)^2 - s_{12}}) / 2}^{\qty(E_\phi - \rmk_h + \sqrt{\qty(E_\phi - \rmk_h)^2 - s_{12}}) / 2} \dd{\rmk_1} ~ \sumAmpNormSqr{\phi \to \varphi \varphi h} \\
        & & & & & \qquad {} \times \qty[1 + f_\varphi\qty(t, \rmk_1) + f_\varphi\qty(t, E_\phi - \rmk_h - \rmk_1) + f_\varphi\qty(t, \rmk_1) f_\varphi\qty(t, E_\phi - \rmk_h - \rmk_1)] \\
        & & = & & & \frac{n_\phi(t)}{128 \twopi[] \rmk_h^2} \int_0^{m_\phi^2} \frac{\dd{s_{12}}}{\sqrt{\qty(\sqrt{m_\phi^2 + \epsilon^2} - \rmk_h)^2 - s_{12}}} \frac{\epsilon}{\sqrt{m_\phi^2 + \epsilon^2}} \frac{1}{\epsilon^2} \\
        & & & & & \qquad {} \times \tilde{\Theta}\qty(\abs{\frac{\rmk_h m_\phi^2}{m_\phi^2 - s_{12}} - \frac{m_\phi^2 - s_{12}}{4 \rmk_h}} < \epsilon) \\
        & & & & & \qquad {} \times \int_{\qty(\sqrt{m_\phi^2 + \epsilon^2} - \rmk_h - \sqrt{\qty(\sqrt{m_\phi^2 + \epsilon^2} - \rmk_h)^2 - s_{12}}) / 2}^{\qty(\sqrt{m_\phi^2 + \epsilon^2} - \rmk_h + \sqrt{\qty(\sqrt{m_\phi^2 + \epsilon^2} - \rmk_h)^2 - s_{12}}) / 2} \dd{\rmk_1} ~ \sumAmpNormSqr{\phi \to \varphi \varphi h} \\
        & & & & & \qquad {} \times \qty[
            \begin{aligned}
                1 + f_\varphi\qty(t, \rmk_1) + f_\varphi\qty(t, \sqrt{m_\phi^2 + \epsilon^2} - \rmk_h - \rmk_1) \\
                {} + f_\varphi\qty(t, \rmk_1) f_\varphi\qty(t, \sqrt{m_\phi^2 + \epsilon^2} - \rmk_h - \rmk_1)
            \end{aligned}
        ] \\
        & & \tensor[_{\epsilon \to 0^+}]{=}{} & & & \frac{n_\phi(t)}{128 \twopi[] \rmk_h^2} \int_0^{m_\phi^2} \frac{2 \dd{s_{12}}}{m_\phi \sqrt{\qty(m_\phi - \rmk_h)^2 - s_{12}}} \delta\qty(\frac{\rmk_h m_\phi^2}{m_\phi^2 - s_{12}} - \frac{m_\phi^2 - s_{12}}{4 \rmk_h}) \\
        & & & & & \qquad {} \times \int_{\qty(m_\phi - \rmk_h - \sqrt{\qty(m_\phi - \rmk_h)^2 - s_{12}}) / 2}^{\qty(m_\phi - \rmk_h + \sqrt{\qty(m_\phi - \rmk_h)^2 - s_{12}}) / 2} \dd{\rmk_1} ~ \sumAmpNormSqr{\phi \to \varphi \varphi h} \\
        & & & & & \qquad {} \times \qty[1 + f_\varphi\qty(t, \rmk_1) + f_\varphi\qty(t, m_\phi - \rmk_h - \rmk_1) + f_\varphi\qty(t, \rmk_1) f_\varphi\qty(t, m_\phi - \rmk_h - \rmk_1)] \\
        & & = & & & \frac{n_\phi(t)}{32 \twopi[] m_\phi \rmk_h^2} \tilde{\Theta}\qty(\rmk_h \le \frac{m_\phi}{2}) \int_{m_\phi / 2 - \rmk_h}^{m_\phi / 2} \dd{\rmk_1} ~ \sumAmpNormSqr{\phi \to \varphi \varphi h} \\
        & & & & & \qquad {} \times \qty[1 + f_\varphi\qty(t, \rmk_1) + f_\varphi\qty(t, m_\phi - \rmk_h - \rmk_1) + f_\varphi\qty(t, \rmk_1) f_\varphi\qty(t, m_\phi - \rmk_h - \rmk_1)] \\
    \end{aligned}
\end{equation}
Then applying $f_\varphi$, we have
\begin{equation}
    \begin{aligned}
        & & & \CT[f_h][\phi \leftrightarrow \varphi \varphi h][t, \rmk_h] \\
        & \simeq & & \frac{n_\phi(t)}{32 \twopi[] m_\phi \rmk_h^2} \tilde{\Theta}\qty(\rmk_h \le \frac{m_\phi}{2}) \int_{m_\phi / 2 - \rmk_h}^{m_\phi / 2} \dd{\rmk_1} ~ \sumAmpNormSqr{\phi \to \varphi \varphi h} \\
        & & & \qquad {} \times \qty[
            \begin{aligned}
                1 + \frac{2 \pi^2}{\rmk_1^2} n_\varphi(t) \delta\qty(\rmk_1 - \frac{m_\phi}{2} \frac{a_\ini}{a(t)}) \\
                {} + \frac{2 \pi^2}{(m_\phi - \rmk_h - \rmk_1)^2} n_\varphi(t) \delta\qty(m_\phi - \rmk_h - \rmk_1 - \frac{m_\phi}{2} \frac{a_\ini}{a(t)}) \\
                {} + \frac{4 \pi^4}{\rmk_1^2 (m_\phi - \rmk_h - \rmk_1)^2} n_\varphi^2(t) \delta\qty(\rmk_1 - \frac{m_\phi}{2} \frac{a_\ini}{a(t)}) \delta\qty(m_\phi - \rmk_h - \rmk_1 - \frac{m_\phi}{2} \frac{a_\ini}{a(t)})
            \end{aligned}
        ] \\
        & = & & \frac{n_\phi(t)}{32 \twopi[] m_\phi \rmk_h^2} \tilde{\Theta}\qty(\rmk_h \le \frac{m_\phi}{2}) \sumAmpNormSqr{\phi \to \varphi \varphi h} \\
        & & & \qquad {} \times \qty[
            \begin{aligned}
                \rmk_h + 2 \pi^2 n_\varphi(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^2 \tilde{\Theta}\qty(\frac{m_\phi}{2} \frac{a_\ini}{a(t)} \ge \frac{m_\phi}{2} - \rmk_h) \\
                {} + 2 \pi^2 n_\varphi(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^2 \tilde{\Theta}\qty(m_\phi - \rmk_h - \frac{m_\phi}{2} \frac{a_\ini}{a(t)} \le \frac{m_\phi}{2}) \\
                {} + 4 \pi^4 n_\varphi^2(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^4 \delta\qty(m_\phi - \rmk_h - \frac{m_\phi a_\ini}{a(t)}) \tilde{\Theta}\qty(\frac{m_\phi}{2} \frac{a_\ini}{a(t)} \ge \frac{m_\phi}{2} - \rmk_h)
            \end{aligned}
        ] \\
        & = & & \frac{n_\phi(t)}{32 \twopi[] m_\phi \rmk_h^2} \tilde{\Theta}\qty(\rmk_h \le \frac{m_\phi}{2}) \sumAmpNormSqr{\phi \to \varphi \varphi h} \\
        & & & \qquad {} \times \qty[
            \begin{aligned}
                \rmk_h + 4 \pi^2 n_\varphi(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^2 \tilde{\Theta}\qty(\rmk_h \ge \frac{m_\phi}{2} \qty[1 - \frac{a_\ini}{a(t)}]) \\
                {} + 4 \pi^4 n_\varphi^2(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^4 \delta\qty(\rmk_h - m_\phi \qty[1 - \frac{a_\ini}{a(t)}]) \tilde{\Theta}\qty(\rmk_h \ge \frac{m_\phi}{2} \qty[1 - \frac{a_\ini}{a(t)}])
            \end{aligned}
        ] \\
        & = & & \frac{n_\phi(t)}{32 \twopi[] m_\phi \rmk_h^2} \tilde{\Theta}\qty(\rmk_h \le \frac{m_\phi}{2}) \sumAmpNormSqr{\phi \to \varphi \varphi h} \\
        & & & \qquad {} \times \qty[
            \begin{aligned}
                \rmk_h + 4 \pi^2 n_\varphi(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^2 \tilde{\Theta}\qty(\rmk_h \ge \frac{m_\phi}{2} \qty[1 - \frac{a_\ini}{a(t)}]) \\
                {} + 4 \pi^4 n_\varphi^2(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^4 \delta\qty(\rmk_h - m_\phi \qty[1 - \frac{a_\ini}{a(t)}])
            \end{aligned}
        ] \\
        & = & & \frac{n_\phi(t)}{32 \twopi[] m_\phi \rmk_h^2} \tilde{\Theta}\qty(\rmk_h \le \frac{m_\phi}{2}) \frac{2 \lambda^2}{\MPl^2} \qty(1 - \frac{m_\phi}{2 \rmk_h})^2 \\
        & & & \qquad {} \times \qty[
            \begin{aligned}
                \rmk_h + 4 \pi^2 n_\varphi(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^2 \tilde{\Theta}\qty(\rmk_h \ge \frac{m_\phi}{2} \qty[1 - \frac{a_\ini}{a(t)}]) \\
                {} + 4 \pi^4 n_\varphi^2(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^4 \delta\qty(\rmk_h - m_\phi \qty[1 - \frac{a_\ini}{a(t)}])
            \end{aligned}
        ].
    \end{aligned}
\end{equation}
With $\qty(t, \rmk_h) \to \qty(a, \tilde{\rmk}_h \coloneq \rmk_h a)$, we have
\begin{equation}
    \begin{aligned}
        \CT[f_h][\phi \leftrightarrow \varphi \varphi h][a, \tilde{\rmk}_h] & = & & \frac{n_\phi(a) a^2}{32 \twopi[] m_\phi \tilde{\rmk}_h^2} \tilde{\Theta}\qty(\tilde{\rmk}_h \le \frac{m_\phi a}{2}) \frac{2 \lambda^2}{\MPl^2} \qty(1 - \frac{m_\phi a}{2 \tilde{\rmk}_h})^2 \\
        & & & \qquad {} \times \qty[
            \begin{aligned}
                \frac{\tilde{\rmk}_h}{a} + 4 \pi^2 n_\varphi(a) \qty[\frac{2 a}{m_\phi a_\ini}]^2 \tilde{\Theta}\qty(\tilde{\rmk}_h \ge \frac{m_\phi a}{2} \qty[1 - \frac{a_\ini}{a}]) \\
                {} + 4 \pi^4 n_\varphi^2(a) \qty[\frac{2 a}{m_\phi a_\ini}]^4 \delta\qty(\tilde{\rmk}_h - m_\phi a \qty[1 - \frac{a_\ini}{a}])
            \end{aligned}
        ].
    \end{aligned}
\end{equation}
Therefore,
\begin{equation}
    \begin{aligned}
        \tilde{f}_h\qty(a, \tilde{\rmk}_h) & = & & \int_{a_\ini}^a \frac{\dd{\alpha}}{H(\alpha) \alpha} \CT[f_h][\phi \leftrightarrow \varphi \varphi h][\alpha, \tilde{\rmk}_h] \\
        & = & & \int_{a_\ini}^a \frac{\dd{\alpha}}{H(\alpha) \alpha} \frac{n_\phi(\alpha) \alpha^2}{32 \twopi[] m_\phi \tilde{\rmk}_h^2} \tilde{\Theta}\qty(\tilde{\rmk}_h \le \frac{m_\phi \alpha}{2}) \frac{2 \lambda^2}{\MPl^2} \qty(1 - \frac{m_\phi \alpha}{2 \tilde{\rmk}_h})^2 \\
        & & & \qquad {} \times \qty[
            \begin{aligned}
                \frac{\tilde{\rmk}_h}{\alpha} + 4 \pi^2 n_\varphi(\alpha) \qty[\frac{2 \alpha}{m_\phi a_\ini}]^2 \tilde{\Theta}\qty(\tilde{\rmk}_h \ge \frac{m_\phi \alpha}{2} \qty[1 - \frac{a_\ini}{\alpha}]) \\
                {} + 4 \pi^4 n_\varphi^2(\alpha) \qty[\frac{2 \alpha}{m_\phi a_\ini}]^4 \delta\qty(\tilde{\rmk}_h - m_\phi \alpha \qty[1 - \frac{a_\ini}{\alpha}])
            \end{aligned}
        ] \\
        & = & & \tilde{\Theta}\qty(2 \tilde{\rmk}_h \le m_\phi a) \int_{\max\qty(a_\ini, 2 \tilde{\rmk}_h / m_\phi)}^a \frac{\dd{\alpha}}{H(\alpha) \alpha} \frac{n_\phi(\alpha) \alpha^2}{32 \twopi[] m_\phi \tilde{\rmk}_h^2} \frac{2 \lambda^2}{\MPl^2} \qty(1 - \frac{m_\phi \alpha}{2 \tilde{\rmk}_h})^2 \\
        & & & \qquad {} \times \qty[
            \begin{aligned}
                \frac{\tilde{\rmk}_h}{\alpha} + 4 \pi^2 n_\varphi(\alpha) \qty[\frac{2 \alpha}{m_\phi a_\ini}]^2 \tilde{\Theta}\qty(\tilde{\rmk}_h \ge \frac{m_\phi \alpha}{2} \qty[1 - \frac{a_\ini}{\alpha}]) \\
                {} + 4 \pi^4 n_\varphi^2(\alpha) \qty[\frac{2 \alpha}{m_\phi a_\ini}]^4 \delta\qty(\tilde{\rmk}_h - m_\phi \alpha \qty[1 - \frac{a_\ini}{\alpha}])
            \end{aligned}
        ].
    \end{aligned}
\end{equation}
Concentrating on the Dirac-$\delta$ term, we have
\begin{equation}
    \int_{\max\qty(a_\ini, 2 \tilde{\rmk}_h / m_\phi)}^a \dd{\alpha} \delta\qty(\tilde{\rmk}_h - m_\phi \alpha \qty[1 - \frac{a_\ini}{\alpha}]) = m_\phi \tilde{\Theta}\qty(\tilde{\rmk}_h \le m_\phi a_\ini) \tilde{\Theta}\qty(\tilde{\rmk}_h \le m_\phi \qty(a - a_\ini))
\end{equation}

\subsection{\boldmath For \texorpdfstring{$f_\phi$}{f\_phi}}

For $f_\phi$, the collision term reads
\begin{equation}
    \CT[f_\phi][\phi \leftrightarrow \varphi \varphi h][t, \rmp_\phi] = \frac{1}{2 E_\phi} \int \tildedp{k_1} ~ \tildedp{k_2} ~ \tildedp{k_h} ~ \twopidelta[4]{p_1 - k_1 - k_2 - k_h} ~ \frac{1}{2} \sumAmpNormSqr{\phi \to \varphi \varphi h} F_-\qty(t, \rmp_\phi, \rmk_1, \rmk_2)
\end{equation}
where the factor of $1 / 2$ is the symmetry factor for two identical particles $\varphi$ in the final state, and norm squared of amplitude that summed over the left- and right-hand side degrees of freedom is given by
\begin{equation}
    \sumAmpNormSqr{\phi \to \varphi \varphi h} = \frac{2 \lambda^2}{\MPl^2} \qty(1 - \frac{m_\phi}{2 E_h})^2.
\end{equation}

Assuming that the non-relativistic limit of $\phi$ is always valid, i.e., $f_\phi\qty(t, \rmp_\phi) \propto \delta\qty(\rmp_\phi - \epsilon)$ with $\epsilon \to 0^+$ as the initial condition in Eq.~\eqref{eq:initial-condition-for-f_phi}, we can perform the integration in the rest frame of $\phi$ as \cite[Eqs.~(V.2.10) and (V.2.14)]{Byckling:1973vca}
\begin{equation}
    \int \tildedp{k_1} ~ \tildedp{k_2} ~ \tildedp{k_h} ~ \twopidelta[4]{p_\phi - k_1 - k_2 - k_h} = \frac{1}{16 \twopi[3] s} \int_0^{m_\phi^2} \dd{s_{1h}} \int_0^{m_\phi^2 - s_{1h}} \dd{s_{2h}},
\end{equation}
where the kinematic invariants are defined by
\begin{align}
    s & \coloneq -p_\phi^2 = m_\phi^2, \\
    s_{1h} & \coloneq -\qty(k_1 + k_h)^2, \\
    s_{2h} & \coloneq -\qty(k_2 + k_h)^2.
\end{align}
According to Eq.~(V.1.3) of Ref.~\cite{Byckling:1973vca}, we also have
\begin{align}
    \rmk_1 & = \frac{m_\phi^2 - s_{2h}}{2 m_\phi}, \\
    \rmk_2 & = \frac{m_\phi^2 - s_{1h}}{2 m_\phi}, \\
    \rmk_h & = \frac{s_{1h} + s_{2h}}{2 m_\phi}.
\end{align}
Therefore, the collision term can be rewritten as
\begin{equation}
    \CT[f_\phi][\phi \leftrightarrow \varphi \varphi h][t, \rmp_\phi] \simeq \frac{\delta\qty(\rmp_\phi - 0^+)}{64 \twopi[3] m_\phi^3} \int_0^{m_\phi^2} \dd{s_{1h}} \int_0^{m_\phi^2 - s_{1h}} \dd{s_{2h}} ~ \sumAmpNormSqr{\phi \to \varphi \varphi h} F_-\qty(t, \rmp_\phi, \rmk_1, \rmk_2).
\end{equation}
Integrating it, we have
\begin{equation}
    \begin{aligned}
        \int \frac{\dd[3]{\vb{p}_\phi}}{\twopi[3]} \CT[f_\phi][\phi \leftrightarrow \varphi \varphi h][t, \rmp_\phi] & = & & \frac{n_\phi(t)}{64 \twopi[3] m_\phi^3} \int_0^{m_\phi^2} \dd{s_{1h}} \int_0^{m_\phi^2 - s_{1h}} \dd{s_{2h}} \frac{2 \lambda^2}{\MPl^2} \qty(1 - \frac{m_\phi^2}{s_{1h} + s_{2h}})^2 \\
        & & & \qquad{} \times \qty[1 + 2 f_\varphi\qty(t, \frac{m_\phi^2 - s_{2h}}{2 m_\phi}) + f_\varphi\qty(t, \frac{m_\phi^2 - s_{1h}}{2 m_\phi}) f_\varphi\qty(t, \frac{m_\phi^2 - s_{2h}}{2 m_\phi})].
    \end{aligned}
\end{equation}
With $s_\mathrm{P} \equiv s_{1h} + s_{2h}$ and $s_\mathrm{M} \equiv s_{1h} - s_{2h}$, we have
\begin{equation}
    \begin{aligned}
        \int \frac{\dd[3]{\vb{p}_\phi}}{\twopi[3]} \CT[f_\phi][\phi \leftrightarrow \varphi \varphi h][t, \rmp_\phi] & = & & \frac{n_\phi(t)}{16 \twopi[3] m_\phi^3} \frac{\lambda^2}{\MPl^2} \int_0^{m_\phi^2} \dd{s_\mathrm{P}} \qty(1 - \frac{m_\phi^2}{s_\mathrm{P}})^2 \int_{-s_\mathrm{P}}^{s_\mathrm{P}} \dd{s_\mathrm{M}} \\
        & & & \qquad{} \times \qty[
            \begin{aligned}
                1 + 2 f_\varphi\qty(t, \frac{2 m_\phi^2 - s_\mathrm{P} + s_\mathrm{M}}{4 m_\phi}) \\
                {} + f_\varphi\qty(t, \frac{2 m_\phi^2 - s_\mathrm{P} + s_\mathrm{M}}{4 m_\phi}) f_\varphi\qty(t, \frac{2 m_\phi^2 - s_\mathrm{P} - s_\mathrm{M}}{4 m_\phi})
            \end{aligned}
        ].
    \end{aligned}
\end{equation}
Noticing that the infrared divergence appears when $\rmk_h \to 0$, i.e., $s_\mathrm{P} \to 0$, we now introduce $0 < \delta_s < 1$ to regularize the divergence as
\begin{equation}
    \int_0^{m_\phi^2} \dd{s_\mathrm{P}} \qty(1 - \frac{m_\phi^2}{s_\mathrm{P}})^2 \int_{-s_\mathrm{P}}^{s_\mathrm{P}} \dd{s_\mathrm{M}} \to \int_{\delta_s m_\phi^2}^{m_\phi^2} \dd{s_\mathrm{P}} \qty(1 - \frac{m_\phi^2}{s_\mathrm{P}})^2 \int_{-s_\mathrm{P}}^{s_\mathrm{P}} \dd{s_\mathrm{M}}.
\end{equation}

Now we consider the assumption of
\begin{equation}
    f_\varphi\qty(t, \rmp_\varphi) = \frac{2 \pi^2}{\rmp_\varphi^2} n_\varphi(t) \delta\qty(\rmp_\varphi - \frac{m_\phi}{2} \frac{a_\ini}{a(t)}).
\end{equation}
Concentrating on the term of $f_\varphi f_\varphi$, it is non-vanishing only if
\begin{equation*}
    \delta_s m_\phi^2 \le 2 m_\phi^2 \qty[1 - \frac{a_\ini}{a(t)}] \le m_\phi^2,
\end{equation*}
or equivalently,
\begin{equation*}
    0 \le \frac{a(t)}{a_\ini} \le 2 \qand \delta_s \le 2 \qty[1 - \frac{a_\ini}{a(t)}].
\end{equation*}
So we require that
\begin{equation}
    \delta_s \ge 2 \qty[1 - \frac{a_\ini}{a(t)}].
\end{equation}
to make sure that the term of $f_\varphi f_\varphi$ is always vanishing.
Therefore,
\begin{equation}
    \begin{aligned}
        & & & \int \frac{\dd[3]{\vb{p}_\phi}}{\twopi[3]} \CT[f_\phi][\phi \leftrightarrow \varphi \varphi h][t, \rmp_\phi] \\
        & = & & \frac{n_\phi(t)}{16 \twopi[3] m_\phi^3} \frac{\lambda^2}{\MPl^2} \int_{\delta_s m_\phi^2}^{m_\phi^2} \dd{s_\mathrm{P}} \qty(1 - \frac{m_\phi^2}{s_\mathrm{P}})^2 \qty[2 s_\mathrm{P} + 4 \pi^2 n_\varphi(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^2 4 m_\phi \tilde{\Theta}\qty(s_\mathrm{P} \ge m_\phi^2 \qty[1 - \frac{a_\ini}{a(t)}])] \\
        & = & & \frac{n_\phi(t)}{16 \twopi[3] m_\phi^3} \frac{\lambda^2}{\MPl^2} \qty[-m_\phi^4 \qty(2 \ln \delta_s + 3 - 4 \delta_s + \delta_s^2) + 16 \pi^2 n_\varphi(t) \qty[\frac{2}{m_\phi} \frac{a(t)}{a_\ini}]^2 m_\phi^3 \qty(\delta_s^{-1} + 2 \ln \delta_s - \delta_s)].
    \end{aligned}
\end{equation}

\end{equation}

\begin{equation}
    \begin{aligned}
        n_\phi(a) \times \qty(\frac{a}{a_\ini})^3 & \simeq \min\qty(\frac{3 \MPl^2 H_\ini^2}{m_\phi}, \frac{m_\phi^4 H_\ini}{\pi \lambda^2} \ln\frac{32 \pi m_\phi H_\ini}{\lambda^2}) \exp[-\Gamma_{\phi \to \varphi \varphi} (t - t_\ini)] \\
        & \simeq n_\phi'^{(\ini)} \exp[-\frac{C_H \lambda^2}{32 \pi m_\phi H(a)}],
    \end{aligned}
\end{equation}
where we have applied $t - t_\ini \simeq t$ for $t \gg t_\ini$ and $H = C_H / t$ with $C_H = 2/3$ for the matter-dominated (MD) era or $C_H = 1/2$ during the radiation-dominated (RD) era.
And we have defined that
\begin{equation}
    n_\phi'^{(\ini)} \equiv \min\qty(\frac{3 \MPl^2 H_\ini^2}{m_\phi}, \frac{m_\phi^4 H_\ini}{\pi \lambda^2} \ln\frac{32 \pi m_\phi H_\ini}{\lambda^2}).
\end{equation}
If the evolution history of the Universe is purely MD or RD, the Hubble parameter can be simply written as
\begin{equation}
    H(a) = H_\ini \qty(\frac{a_\ini}{a})^{C_H^{-1}}.
\end{equation}
Therefore,
\begin{equation}
    \begin{aligned}
        f_\varphi(a, \rmp_\varphi) & \simeq \exp{\frac{\pi \lambda^2 n_\phi'^{(\ini)}}{m_\phi^4 H_\ini} \qty(\frac{a_\ini}{a'})^{3 - C_H^{-1}} \exp[-\frac{C_H \lambda^2}{32 \pi m_\phi H_\ini} \qty(\frac{a'}{a_\ini})^{C_H^{-1}}] \tilde{\Theta}\qty(a_\ini \le a' \le a)}_{a' = 2 \rmp_\varphi a / m_\phi} - 1 \\
        & =  \exp{\frac{\pi \lambda^2 n_\phi'^{(\ini)}}{m_\phi^4 H_\ini} \qty(\frac{m_\phi a_\ini}{2 \rmp_\varphi a})^{3 - C_H^{-1}} \exp[-\frac{C_H \lambda^2}{32 \pi m_\phi H_\ini} \qty(\frac{2 \rmp_\varphi a}{m_\phi a_\ini})^{C_H^{-1}}] \tilde{\Theta}\qty(\frac{m_\phi}{2} \frac{a_\ini}{a} \le \rmp_\varphi \le \frac{m_\phi}{2})} - 1.
    \end{aligned}
\end{equation}

\section{Calculations}

First of all, we insert two identities of
\begin{align}
    1 & \equiv \int \frac{\dd[4]{q}}{\twopi[4]} \twopidelta[4]{q - p_1 - p_2}, \\
    1 & \equiv \int \frac{\dd{s}}{\twopi} \twopidelta{s + q^2}
\end{align}
into the integration of $\CT[f_h][\varphi \varphi \to h h]$ to obtain
\begin{equation}
    \begin{aligned}
        & & & \CT[f_h][\varphi \varphi \to h h] \\
        & = & & \frac{1}{4 E_1^{(h)}} \int \frac{\dd{s}}{\twopi} \twopidelta{s + q^2} \int \frac{\dd[4]{q}}{\twopi[3]} \twopidelta[4]{q - p_1 - p_2} \int \tildedp{p_1} ~ \tildedp{p_2} ~ \tildedp{k_2} ~ \twopidelta[4]{q - k_1 - k_2} \\
        & & & \qquad {} \times \sumAmpNormSqr{\varphi(p_1) + \varphi(p_2) \to h(k_1) + h(k_2)} f_\varphi(p_1) f_\varphi(p_2) \\
        & = & & \frac{1}{4 E_1^{(h)}} \int_0^\infty \frac{\dd{s}}{\twopi} \int \tildedp{q} \int \tildedp{p_1} ~ \tildedp{p_2} ~ \twopidelta[4]{q - p_1 - p_2} f_\varphi(p_1) f_\varphi(p_2) \\
        & & & \qquad {} \times \int \tildedp{k_2} ~ \twopidelta[4]{q - k_1 - k_2} \frac{2 t^2 \qty(s + t)^2}{\MPl^4 s^2}. \\
    \end{aligned}
\end{equation}
Since $m_h = 0$, we have $E_1^{(h)} = \rmk_1$, and
\begin{equation}
    s = 2 \rmp_\phi \rmp_2 \qty(1 - \cos \expval{\vb{p}_1, \vb{p}_2}) = 2 \rmk_1 \rmk_2 \qty(1 - \cos \expval{\vb{k}_1, \vb{k}_2}),
\end{equation}
which leads to
\begin{equation}
    \begin{aligned}
        & & & \frac{1}{4 \rmk_1} \int_0^\infty \frac{\dd{s}}{\twopi} \int \tildedp{q} \int \tildedp{k_2} ~ \twopidelta[4]{q - k_1 - k_2} \\
        & = & & \frac{1}{4 \rmk_1} \int_0^\infty \frac{\dd{s}}{\twopi} \int \tildedp{q} ~ \frac{1}{2 \rmk_2} \twopidelta{\sqrt{\rmq^2 + s} - \rmk_1 - \rmk_2} \\
        & = & & \frac{1}{4 \rmk_1} \int_0^\infty \frac{\dd{s}}{\twopi} \int \frac{\dd[3]{\vb{q}}}{\twopi[3] 2 \sqrt{\rmq^2 + s}} \frac{\rmk_1 \qty(1 - \cos \expval{\vb{k}_1, \vb{k}_2})}{s} \\
        & & & \qquad {} \times \twopidelta{\sqrt{\rmq^2 + s} - \rmk_1 - \frac{s}{2 \rmk_1 \qty(1 - \cos \expval{\vb{k}_1, \vb{k}_2})}} \\
        & = & & \frac{1}{4} \int_0^\infty \frac{\dd{s}}{\twopi s} \int_{-1}^{+1} \dd{\cos \expval{\vb{q}, \vb{k}_1}} \int_0^\infty \frac{\rmq^2 \dd{\rmq}}{\twopi[2] 2 \sqrt{\rmq^2 + s}} \qty(1 - \cos \expval{\vb{k}_1, \vb{k}_2}) \\
        & & & \qquad {} \times \twopidelta{\sqrt{\rmq^2 + s} - \rmk_1 - \frac{s}{2 \rmk_1 \qty(1 - \cos \expval{\vb{k}_1, \vb{k}_2})}}.
    \end{aligned}
\end{equation}
Noticing $\vb{q} = \vb{k}_1 + \vb{k}_2$, we have
\begin{equation}
    \begin{aligned}
        \vb{q} \cdot \vb{k}_1 & = \vb{k}_1^2 + \vb{k}_1 \cdot \vb{k}_2 \\
        \rmq \rmk_1 \cos \expval{\vb{q}, \vb{k}_1} & = \rmk_1^2 + \rmk_1 \rmk_2 \cos \expval{\vb{k}_1, \vb{k}_2} \\
        & = \rmk_1^2 + \frac{s \cos \expval{\vb{k}_1, \vb{k}_2}}{2 (1 - \cos \expval{\vb{k}_1, \vb{k}_2})},
    \end{aligned}
\end{equation}
where we have applied
\begin{equation*}
    \rmk_2 = \frac{s}{2 \rmk_1 (1 - \cos \expval{\vb{k}_1, \vb{k}_2})}.
\end{equation*}
Therefore,
\begin{equation}
    \begin{aligned}
        & & & \frac{1}{4} \int_0^\infty \frac{\dd{s}}{\twopi s} \int_{-1}^{+1} \dd{\cos \expval{\vb{q}, \vb{k}_1}} \int_0^\infty \frac{\rmq^2 \dd{\rmq}}{\twopi[2] 2 \sqrt{\rmq^2 + s}} \qty(1 - \cos \expval{\vb{k}_1, \vb{k}_2}) \\
        & & & \qquad {} \times \twopidelta{\sqrt{\rmq^2 + s} - \rmk_1 - \frac{s}{2 \rmk_1 \qty(1 - \cos \expval{\vb{k}_1, \vb{k}_2})}} \\
        & = & & \frac{1}{4} \int_0^\infty \frac{\dd{s}}{\twopi s} \int_0^\infty \frac{\rmq^2 \dd{\rmq}}{\twopi[2] 2 \sqrt{\rmq^2 + s}} \int_{-1}^{+1} \dd{\cos \expval{\vb{q}, \vb{k}_1}} \frac{s}{s - 2 \rmk_1 \qty(\rmk_1 - q \cos \expval{\vb{q}, \vb{k}_1})} \\
        & & & \qquad {} \times \twopidelta{\sqrt{\rmq^2 + s} - \frac{s}{2 \rmk_1} - \rmq \cos \expval{\vb{q}, \vb{k}_1}} \\
        & = & & \frac{1}{4} \int_0^\infty \frac{\dd{s}}{\twopi s} \int_0^\infty \frac{\rmq^2 \dd{\rmq}}{\twopi[2] 2 \sqrt{\rmq^2 + s}} \frac{\twopi s}{2 \rmq \rmk_1 \qty(\sqrt{\rmq^2 + s} - \rmk_1)} \tilde{\Theta}\qty(q \ge \frac{\abs{s - 4 \rmk_1^2}}{4 \rmk_1}) \\
        & = & & \frac{1}{16 \twopi[2] \rmk_1} \int_0^\infty \dd{s} \int_{\abs{s - 4 \rmk_1^2} / (4 \rmk_1)}^\infty \frac{\rmq \dd{\rmq}}{\rmq^2 + s - \rmk_1 \sqrt{\rmq^2 + s}} \\
        & = & & \frac{1}{16 \twopi[2] \rmk_1} \int_0^\infty \dd{s} \int_{\rmk_1 + s / (4 \rmk_1)}^\infty \frac{\dd{E_q}}{E_q - \rmk_1} \\
        & = & & \frac{1}{16 \twopi[2] \rmk_1} \int_{\rmk_1}^\infty \frac{\dd{E_q}}{E_q - \rmk_1} \int_0^{4 \rmk_1 (E_q - \rmk_1)} \dd{s},
    \end{aligned}
\end{equation}
where we have applied $E_q = \sqrt{\rmq^2 + s}$.
Taking it back into the integration of $\CT[f_h][\varphi \varphi \to h h]$, we have
\begin{equation}
    \begin{aligned}
        & & & \CT[f_h][\varphi \varphi \to h h] \\
        & = & & \frac{1}{16 \twopi[2] \rmk_1} \int_{\rmk_1}^\infty \frac{\dd{E_q}}{E_q - \rmk_1} \int_0^{4 \rmk_1 (E_q - \rmk_1)} \dd{s} \\
        & & & \qquad {} \times \int \tildedp{p_1} ~ \tildedp{p_2} ~ \twopidelta[4]{q - p_1 - p_2} f_\varphi(p_1) f_\varphi(p_2) \frac{2 t^2 \qty(s + t)^2}{\MPl^4 s^2}. \\
    \end{aligned}
\end{equation}
Then, we can continue to calculate the integration over $p_1$ and $p_2$ as
\begin{equation}
    \begin{aligned}
        \int \tildedp{p_1} ~ \tildedp{p_2} ~ \twopidelta[4]{q - p_1 - p_2} & = & & \int \frac{\tildedp{p_1}}{2 E_2^{(\varphi)}} \twopidelta{E_q - E_1^{(\varphi)} - E_2^{(\varphi)}} \\
        & = & & \int \frac{\tildedp{p_1}}{2 \rmp_2} \twopidelta{E_q - \rmp_\phi - \rmp_2} \\
        & = & & \int_0^\infty \frac{\rmp_\phi \dd{\rmp_\phi}}{\twopi[3] 4 \rmp_2} \int_{-1}^{+1} \cos \expval{\vb{q}, \vb{p}_1} \int_{0}^{2 \pi} \dd{\varphi} \\
        & & & \qquad {} \times \twopidelta{E_q - \rmp_\phi - \sqrt{\rmq^2 + \rmp_\phi^2 - 2 \rmq \rmp_\phi \cos \expval{\vb{q}, \vb{p}_1}}} \\
        & = & & \frac{1}{4 \twopi[2]} \int_{0}^{2 \pi} \dd{\varphi} \int_0^\infty \frac{\rmp_\phi \dd{\rmp_\phi}}{\rmp_2} \frac{\rmp_2}{\rmq \rmp_\phi} \tilde{\Theta}\qty(\abs{\rmp_\phi - \frac{E_q}{2}} \le \frac{\rmq}{2}) \\
        & = & & \frac{1}{4 \twopi[2] \rmq} \int_{0}^{2 \pi} \dd{\varphi} \int_{(E_q - \rmq) / 2}^{(E_q + \rmq) / 2} \dd{\rmp_\phi},
    \end{aligned}
\end{equation}
where the azimuthal angle $\varphi$ is not integrated out since the integrand depends on it, and we will return to it later.
Therefore,
\begin{equation}
    \begin{aligned}
        & & & \CT[f_h][\varphi \varphi \to h h] \\
        & = & & \frac{1}{64 \twopi[4] \rmk_1} \int_{\rmk_1}^\infty \frac{\dd{E_q}}{E_q - \rmk_1} \int_0^{4 \rmk_1 (E_q - \rmk_1)} \frac{\dd{s}}{\rmq} \int_{(E_q - \rmq) / 2}^{(E_q + \rmq) / 2} \dd{\rmp_\phi} f_\varphi(\rmp_\phi) f_\varphi\qty(E_q - \rmp_\phi) \\
        & & & \qquad {} \times \int_{0}^{2 \pi} \dd{\varphi} \frac{2 t^2 \qty(s + t)^2}{\MPl^4 s^2} \\
        & = & & \frac{1}{64 \twopi[4] \rmk_1} \int_{\rmk_1}^\infty \frac{\dd{E_q}}{E_q - \rmk_1} \int_0^{E_q} \dd{\rmp_\phi} f_\varphi(\rmp_\phi) f_\varphi\qty(E_q - \rmp_\phi) \int_0^{s_{\max}} \frac{\dd{s}}{\rmq} \\
        & & & \qquad {} \times \int_{0}^{2 \pi} \dd{\varphi} \frac{2 t^2 \qty(s + t)^2}{\MPl^4 s^2},
    \end{aligned}
    \label{eq:CT-f_h-analytical}
\end{equation}
where the Mandelstam variable $t$ is evaluated as
\begin{equation}
    \begin{aligned}
        t \coloneq & -\qty(p_1 - k_1)^2 \\
        = & -2 \rmp_\phi \rmk_1 \qty(1 - \cos \expval{\vb{p}_1, \vb{k}_1})
    \end{aligned}
\end{equation}
and
\begin{equation}
    s_{\max} = 4 \min\qty[\rmk_1 (E_q - \rmk_1), \rmp_\phi \qty(E_q - \rmp_\phi)].
\end{equation}
With spherical law of cosines, we have
\begin{equation}
    \begin{aligned}
        \cos \expval{\vb{p}_1, \vb{k}_1} & = \cos \expval{\vb{q}, \vb{p}_1} \cos \expval{\vb{q}, \vb{k}_1} + \sin \expval{\vb{q}, \vb{p}_1} \sin \expval{\vb{q}, \vb{k}_1} \cos \varphi \\
        & = \cos \expval{\vb{q}, \vb{p}_1} \cos \expval{\vb{q}, \vb{k}_1} + \sqrt{\qty(1 - \cos^2 \expval{\vb{q}, \vb{p}_1}) \qty(1 - \cos^2 \expval{\vb{q}, \vb{k}_1})} \cos \varphi,
    \end{aligned}
\end{equation}
where $\varphi$ is the azimuthal angle between the planes of $(\vb{q}, \vb{p}_1)$ and $(\vb{q}, \vb{k}_1)$.
Notice that $\expval{\vb{q}, \vb{k}_1}$ and $\expval{\vb{q}, \vb{p}_1}$ are in the range of $[0, \pi]$, so
\begin{equation*}
    \sin \expval{\vb{q}, \vb{k}_1} = +\sqrt{1 - \cos^2 \expval{\vb{q}, \vb{k}_1}} \qand \sin \expval{\vb{q}, \vb{p}_1} = + \sqrt{1 - \cos^2 \expval{\vb{q}, \vb{p}_1}}
\end{equation*}
are applied.
There are two cosine terms that have been derived before, and we present them here explicitly as
\begin{align}
    \cos \expval{\vb{q}, \vb{k}_1} & = \frac{2 \rmk_1 E_q - s}{2 \rmk_1 \rmq }, \\
    \cos \expval{\vb{q}, \vb{p}_1} & = \frac{2 \rmp_\phi E_q - s}{2 \rmp_\phi \rmq}.
\end{align}
To evaluate the integration over $\varphi$, we first consider the integration over $t \equiv A + B \cos \varphi$ as
\begin{equation}
    \int_0^{2 \pi} \dd{\varphi} \frac{2 t^2 (s + t)^2}{\MPl^4 s^2} = \frac{\pi}{2 \MPl^4 s^2} \qty[8 A^4 + 3 B^4 + 24 A^2 B^2 + 8 s A (2 A^2 + 3 B^2) + 4 s^2 (2 A^2 + B^2)].
\end{equation}
Finally, we identify that
\begin{align}
    A & \equiv -2 \rmp_\phi \rmk_1 \qty(1 - \cos \expval{\vb{q}, \vb{p}_1} \cos \expval{\vb{q}, \vb{k}_1}), \\
    B & \equiv 2 \rmp_\phi \rmk_1 \sqrt{\qty(1 - \cos^2 \expval{\vb{q}, \vb{p}_1}) \qty(1 - \cos^2 \expval{\vb{q}, \vb{k}_1})},
\end{align}
and the corresponding integral is obtained.
Notice that the integration over $\dd{s}$ can also be performed analytically, but the result is too lengthy to be presented here.

To solve the Boltzmann equation shown in Eq.~\eqref{eq:BTE-for-f_h}, we can consider the change of variables as
\begin{equation}
    \begin{aligned}
        t & \to a(t), \\
        p_h & \to \tilde{p}_h \equiv p_h a(t),
    \end{aligned}
\end{equation}
The Boltzmann equation now becomes
\begin{equation}
    H a \pdv{\tilde{f}_h}{a}\qty(a, \tilde{p}_h) = \CT[f_h][][a, \tilde{p}_h],
\end{equation}
whose solution is simply
\begin{equation}
    \tilde{f}_h\qty(a, \tilde{p}_h) = \int_{a_\ini}^{a} \frac{\dd{\alpha}}{H(\alpha) \alpha} \CT[f_h][][\alpha, \tilde{p}_h].
\end{equation}
Applying the collision term of Eq.~\eqref{eq:CT-f_h-analytical}, we have
\begin{equation}
    \begin{aligned}
        \tilde{f}_h\qty(a, \tilde{k}_1) & = & & \int_{a_\ini}^{a} \frac{\dd{\alpha}}{H(\alpha) \alpha} \frac{1}{64 \twopi[4] \rmk_1} \int_{\rmk_1}^\infty \frac{\dd{E_q}}{E_q - \rmk_1} \int_0^{E_q} \dd{\rmp_\phi} f_\varphi(\rmp_\phi) f_\varphi\qty(E_q - \rmp_\phi) \\
        & & & \qquad {} \times \int_0^{s_{\max}} \frac{\dd{s}}{\rmq} \int_{0}^{2 \pi} \dd{\varphi} \frac{2 t^2 \qty(s + t)^2}{\MPl^4 s^2} \\
        & = & & \frac{1}{64 \twopi[4]} \int_{a_\ini}^{a} \frac{\dd{\alpha}}{H(\alpha) \tilde{\rmk}_1} \int_{\tilde{\rmk}_1}^\infty \frac{\dd{\tilde{E}_q}}{\tilde{E}_q - \tilde{\rmk}_1} \int_0^{\tilde{E}_q} \frac{\dd{\tilde{\rmp}_1}}{\alpha} \tilde{f}_\varphi(\alpha, \tilde{\rmp}_1) \tilde{f}_\varphi\qty(\alpha, \tilde{E}_q - \tilde{\rmp}_1) \\
        & & & \qquad {} \times \int_0^{\tilde{s}_{\max}} \frac{\dd{\tilde{s}}}{\tilde{\rmq} \alpha} \int_{0}^{2 \pi} \dd{\varphi} \frac{2 \tilde{t}^2 \qty(\tilde{s} + \tilde{t})^2}{\MPl^4 \tilde{s}^2 \alpha^4}
    \end{aligned}
\end{equation}
with
\begin{equation}
    \tilde{s}_{\max} = 4 \min\qty[\tilde{\rmk}_1 (\tilde{E}_q - \tilde{\rmk}_1), \tilde{\rmp}_1 (\tilde{E}_q - \tilde{\rmp}_1)],
\end{equation}
where we have also defined that $\tilde{\rmk}_1 \equiv \rmk_1 a$, $\tilde{E}_q \equiv E_q a$, $\tilde{\rmp}_1 \equiv \rmp_\phi a$, $\tilde{s} \equiv s a^2$, $\tilde{\rmq} \equiv \rmq a$, and $\tilde{t} \equiv t a^2$ ($a \to \alpha$ in the integrand).
Here, the phase space distribution of the reheaton $\varphi$ in the comoving momentum is given by
\begin{equation}
    \tilde{f}_\varphi\qty(\alpha, \tilde{\rmp}_\varphi) = \exp[\frac{\pi \lambda^2 n_\phi(\alpha')}{m_\phi^4 H(\alpha')} \tilde{\Theta}\qty(\frac{m_\phi a_\ini}{2} \le \tilde{\rmp}_\varphi \le \frac{m_\phi \alpha}{2})]_{\alpha' = 2 \tilde{\rmp}_\varphi / m_\phi} - 1,
\end{equation}
which requires that
\begin{equation}
    \frac{m_\phi a_\ini}{2} \le \tilde{\rmp}_1 \le \frac{m_\phi \alpha}{2} \qand \frac{m_\phi a_\ini}{2} \le \tilde{E}_q - \tilde{\rmp}_1 \le \frac{m_\phi \alpha}{2}
\end{equation}
for the non-vanishing product of $\tilde{f}_\varphi(\alpha, \tilde{\rmp}_1) \tilde{f}_\varphi(\alpha, \tilde{E}_q - \tilde{\rmp}_1)$.
Summing them up, we have then
\begin{equation}
    m_\phi a_\ini \le \tilde{E}_q \le m_\phi \alpha,
\end{equation}
which also requires $\tilde{\rmk}_1 \le m_\phi \alpha \le m_\phi a$ since $\tilde{\rmk}_1 \le \tilde{E}_q$, \ie,
\begin{equation}
    \max\qty(a_\ini, \frac{\tilde{k}_1}{m_\phi}) \le \alpha \le a.
\end{equation}
Furthermore, the non-vanishing product of $\tilde{f}_\varphi(\alpha, \tilde{\rmp}_1) \tilde{f}_\varphi(\alpha, \tilde{E}_q - \tilde{\rmp}_1)$ also requires
\begin{equation}
    \abs{\tilde{\rmp}_1 - \frac{\tilde{E_q}}{2}} \le \frac{1}{2} \min\qty(m_\phi \alpha - \tilde{E}_q, \tilde{E}_q - m_\phi a_\ini).
\end{equation}
Hence, the integration ranges of $\tilde{\rmk}_1$, $\tilde{\rmq}$ and $\tilde{\rmp}_1$ are constrained, \ie,
\begin{equation}
    \begin{aligned}
        \tilde{f}_h\qty(a, \tilde{\rmk}_1) & = & & \frac{1}{64 \twopi[4] \tilde{\rmk}_1} \tilde{\Theta}\qty(0 \le \tilde{\rmk}_1 \le m_\phi a) \int_{\max\qty(a_\ini, \tilde{k}_1 / m_\phi)}^{a} \frac{\dd{\alpha}}{H(\alpha) \alpha^6} \\
        & & & \qquad {} \times \int_{\max\qty(\tilde{\rmk}_1, m_\phi a_\ini)}^{m_\phi \alpha} \frac{\dd{\tilde{E}_q}}{\tilde{E}_q - \tilde{\rmk}_1} \int_{\tilde{\rmp}_1^{(-)}}^{\tilde{\rmp}_1^{(+)}} \dd{\tilde{\rmp}_1} \tilde{f}_\varphi(\alpha, \tilde{\rmp}_1) \tilde{f}_\varphi\qty(\alpha, \tilde{E}_q - \tilde{\rmp}_1) \\
        & & & \qquad {} \times \int_0^{\tilde{s}_{\max}} \frac{\dd{\tilde{s}}}{\tilde{\rmq}} \int_{0}^{2 \pi} \dd{\varphi} \frac{2 \tilde{t}^2 \qty(\tilde{s} + \tilde{t})^2}{\MPl^4 \tilde{s}^2},
    \end{aligned}
\end{equation}
where
\begin{equation}
    \tilde{\rmp}_1^{(\pm)} = \frac{1}{2} \qty[\tilde{E}_q \pm \min\qty(m_\phi \alpha - \tilde{E}_q, \tilde{E}_q - m_\phi a_\ini)].
\end{equation}
